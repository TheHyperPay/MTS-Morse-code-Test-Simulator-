#include <stdio.h>
#include <windows.h>
#include <string.h>

const char* changeMorse(char alpha) {
    alpha = toupper(alpha);

    switch (alpha) {
    case 'A':
        return ".-";
    case 'B':
        return "-...";
    case 'C':
        return "-.-.";
    case 'D':
        return "-..";
    case 'E':
        return ".";
    case 'F':
        return "..-.";
    case 'G':
        return "--.";
    case 'H':
        return "....";
    case 'I':
        return "..";
    case 'J':
        return ".---";
    case 'K':
        return "-.-";
    case 'L':
        return ".-..";
    case 'M':
        return "--";
    case 'N':
        return "-.";
    case 'O':
        return "---";
    case 'P':
        return ".--.";
    case 'Q':
        return "--.-";
    case 'R':
        return ".-.";
    case 'S':
        return "...";
    case 'T':
        return "-";
    case 'U':
        return "..-";
    case 'V':
        return "...-";
    case 'W':
        return ".--";
    case 'X':
        return "-..-";
    case 'Y':
        return "-.--";
    case 'Z':
        return "--..";
    case ' ':
        return " ";
    default:
        return NULL;
    }
}

void morseSound(const char* message) {
    for (int i = 0; *(message + i) != NULL; i++) {
        const char* morseMessage = changeMorse(*(message + i));

        printf("%c: %s\n", *(message + i), morseMessage);

        for (int j = 0; *(morseMessage + j) != NULL; j++) {
            if (*(morseMessage + i) == '.') {
                Beep(800, 200);
            }
            else if (*(morseMessage + i) == '-') {
                Beep(800, 200);
            }
        }
        
    }
      
}

int main() {
    char message[] = "Run";

    morseSound(message);

    return 0;
}
